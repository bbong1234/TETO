flowchart TD

    %% =========================
    %% 0. 用户输入入口
    %% =========================
    A[用户输入一段话] --> B[点击提交]

    B --> C[系统创建本次输入上下文]
    C --> C1[保留 raw_input 原始输入]
    C --> C2[生成 trace_id 处理链路号]
    C --> C3[生成 input_id 原始输入号]
    C --> C4[记录 submitted_at 提交时间]
    C --> C5[记录 client_time 客户端时间]
    C --> C6[记录 selected_date 当前选中日期]

    %% =========================
    %% 1. 时间轴即时承接
    %% =========================
    C --> D[创建临时时间轴卡片]
    D --> D1[卡片显示原始输入]
    D --> D2[卡片状态 = parsing 解析中]
    D --> D3[此时不是正式记录]
    D3 --> D4[此时没有 record_id]

    %% =========================
    %% 2. 后台预处理
    %% =========================
    C --> E[后台开始解析流水线]

    E --> E1[基础预处理]
    E1 --> E11[清理空白与明显噪声]
    E1 --> E12[识别是否为空输入或明显无效]
    E1 --> E13[识别是否可能为复合句]
    E1 --> E14[识别数字 金额 时间词 地点词]

    E12 --> F{输入是否有效}
    F -- 否 --> F1[时间轴状态更新为 invalid / 需重输]
    F1 --> F2[前端提示无效输入]
    F -- 是 --> G[进入正式清分]

    %% =========================
    %% 3. 第一层：原文骨架层
    %% =========================
    G --> H[第一层清分：原文骨架层]

    H --> H1[保留 raw_input 原文不覆盖]
    H --> H2[提炼 content 主内容]
    H --> H3[判断主类型 type]

    H3 --> H31{主类型判断}
    H31 -- 已发生 --> H32[type = 发生]
    H31 -- 将来要做 --> H33[type = 计划]
    H31 -- 脑中想法 --> H34[type = 想法]
    H31 -- 回顾归纳 --> H35[type = 总结]
    H31 -- 不稳定 --> H36[type = 待确认]

    %% =========================
    %% 4. 第二层：主链事实层
    %% =========================
    H --> I[第二层清分：主链事实层]

    %% 时间组
    I --> I1[时间组解析]
    I1 --> I11[提取 time_text 原始时间表达]
    I1 --> I12[结合 client_time / selected_date 解析时间]
    I1 --> I13[得到 occurred_at 开始时间]
    I1 --> I14[得到 occurred_at_end 结束时间]
    I1 --> I15[得到 duration_minutes 时长]
    I1 --> I16[判断 time_precision 时间精度]

    %% 地点组
    I --> I2[地点组解析]
    I2 --> I21[提取 location 原始地点]
    I2 --> I22[判断 place_type 地点类型]

    %% 事件主干组
    I --> I3[事件主干组解析]
    I3 --> I31[提取 action_text 动作]
    I3 --> I32[提取 event_text 情境/事件]
    I3 --> I33[提取 object_text 指向对象]

    %% 原因结果组
    I --> I4[原因结果组解析]
    I4 --> I41[提取 cause_text 原因]
    I4 --> I42[提取 result 结果文本]
    I4 --> I43[判断 outcome_type 结果类型]
    I4 --> I44[判断 outcome_direction 结果方向]

    %% =========================
    %% 5. 第三层：附属补充层
    %% =========================
    I --> J[第三层清分：附属补充层]

    %% 感受状态组
    J --> J1[感受状态组]
    J1 --> J11[提取 mood 心情]
    J1 --> J12[提取 energy 能量]
    J1 --> J13[提取 status 状态]
    J1 --> J14[提取 body_state 身体状态]

    %% 金额指标组
    J --> J2[金额与指标组]
    J2 --> J21[提取 cost 金额]
    J2 --> J22[判断 money_direction 资金方向]
    J2 --> J23[判断 money_currency 币种]
    J2 --> J24[提取 metric_name 指标名称]
    J2 --> J25[提取 metric_value 指标数值]
    J2 --> J26[提取 metric_unit 指标单位]

    %% 关系协同组
    J --> J3[关系与协同组]
    J3 --> J31[提取 people 关系人]
    J3 --> J32[判断 relation_roles 关系角色]
    J3 --> J33[提取 organization_info 组织信息]

    %% 组织归属补充组
    J --> J4[组织归属补充组]
    J4 --> J41[提取 note 备注]
    J4 --> J42[查找 related_record_ids 可关联记录]
    J4 --> J43[预留 confidence_level 置信度]
    J4 --> J44[预留 review_status 审核状态]

    %% =========================
    %% 6. 复合句拆分
    %% =========================
    J --> K{是否为复合句 / 多动作输入?}

    K -- 否 --> K1[保持单条草稿]
    K -- 是 --> K2[执行拆分]
    K2 --> K21[按动作/事件/时间片段拆成多个候选片段]
    K21 --> K22[每个片段独立补齐主类型 主链事实 附属信息]
    K22 --> K23[形成多条候选 draft]

    %% =========================
    %% 7. 归属判断
    %% =========================
    K1 --> L[归属判断]
    K23 --> L

    L --> L1[归属到主类型]
    L1 --> L2[归属到事项 item]
    L2 --> L3[必要时归属到子项 sub_item]
    L3 --> L4[必要时关联已有记录 related_record_ids]

    L2 --> L21{事项归属是否稳定}
    L21 -- 稳定 --> L22[写入建议 item_id]
    L21 -- 不稳定 --> L23[item_id 待确认]

    L3 --> L31{子项归属是否稳定}
    L31 -- 稳定 --> L32[写入建议 sub_item_id]
    L31 -- 不稳定 --> L33[sub_item_id 可留空]

    %% =========================
    %% 8. 形成结构化确认稿
    %% =========================
    L --> M[生成结构化确认稿 Record Draft]
    M --> M1[第一层：原文骨架]
    M --> M2[第二层：主链事实]
    M --> M3[第三层：附属补充]
    M --> M4[附加建议归属 item_id / sub_item_id]
    M --> M5[附加 confidence_level]
    M --> M6[附加 review_status 初始值]

    %% =========================
    %% 9. 审核门
    %% =========================
    M --> N{审核门 Admission Gate}

    N -- 主链明确且归属稳定 --> N1[结果 = 可直接确认]
    N -- 主链基本成形但关键点不稳 --> N2[结果 = 待用户确认]
    N -- 无法理解或冲突过大 --> N3[结果 = 失败/澄清]

    %% =========================
    %% 10. 前端状态切换
    %% =========================
    N1 --> O[时间轴状态更新为 awaiting_confirmation]
    N2 --> O
    O --> O1[前端展示确认稿]
    O1 --> O2[主确认区：主内容 主类型 时间 地点 主干 原因结果]
    O1 --> O3[折叠补充区：状态 金额 指标 关系人 组织 备注 高级补充]

    N3 --> P[时间轴状态更新为 failed / need_clarification]
    P --> P1[提示用户补充或重输]
    P1 --> P2[结束本次处理]

    %% =========================
    %% 11. 用户确认回流
    %% =========================
    O2 --> Q[用户查看确认稿]
    O3 --> Q

    Q --> Q1{用户操作}
    Q1 -- 直接确认 --> Q2[确认通过]
    Q1 -- 修改字段 --> Q3[用户修改草稿]
    Q1 -- 否决/放弃 --> Q4[拒绝本次草稿]

    Q3 --> Q31[形成修正后的确认稿]
    Q31 --> Q32[必要时重新跑局部归属判断]
    Q32 --> Q2

    Q4 --> Q41[时间轴状态 = rejected / 已放弃]
    Q41 --> Q42[保留 trace 与原始输入用于回溯]
    Q42 --> Q43[结束]

    %% =========================
    %% 12. 正式入库
    %% =========================
    Q2 --> R[确认稿定稿]
    R --> R1[写入正式记录数据]
    R1 --> R2[生成 record_id]
    R2 --> R3[写入 item_id / sub_item_id / related_record_ids]
    R3 --> R4[写入 review_status]
    R4 --> R5[写入 confidence_level]
    R5 --> R6[保留 raw_input 与结构化字段映射]

    %% =========================
    %% 13. 入库后状态与后续消费
    %% =========================
    R6 --> S[时间轴状态更新为 saved]
    S --> S1[前端显示正式记录]
    S1 --> S2[此时 record_id 可见]

    R6 --> T[正式记录可供下游消费]
    T --> T1[事项模块可引用]
    T --> T2[统计模块可读取]
    T --> T3[洞察模块可使用]
    T --> T4[后续关联与回顾可使用]

    %% =========================
    %% 14. 可观测与回溯
    %% =========================
    C --> U[链路追踪与日志]
    E --> U
    M --> U
    R6 --> U

    U --> U1[记录 trace_id]
    U --> U2[记录关键 decision]
    U --> U3[记录错误或失败原因]
    U --> U4[支持后续诊断 回放 纠错]

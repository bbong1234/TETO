# 语义引擎底层结构（P0 + P3）

## Task 1: 新建语义类型定义文件

​**文件**: ﻿src/types/semantic.ts﻿（新建）

按项目规范，独立文件定义语义解析相关类型，然后在 ﻿teto.ts﻿ 中 re-export。

ts

```
// src/types/semantic.ts
// TETO 语义解析引擎 —— 核心类型定义

import type { RecordType } from './teto';

/** 时间锚点 */
export interface TimeAnchor {
  raw: string;                              // 原文时间表达："明天"、"上周三"、"3月15号"
  resolved_date: string;                    // 解析后 ISO 日期："2026-04-21"
  direction: 'past' | 'present' | 'future'; // 时间指向
}

/** 量化指标（从现有 metric 体系继承） */
export interface SemanticMetric {
  value: number;
  unit: string;   // 计量单位（个、次、页、公里）
  name: string;   // 统计对象（单词、俯卧撑、阅读）
}

/** 单条记录的语法结构 */
export interface ParsedSemantic {
  // === 核心断言（主谓宾） ===
  subject: string | null;       // 主语："我"、"爸妈"、"小明"
  action: string;               // 谓语/动作："吃"、"考试"、"跑步"
  object: string | null;        // 宾语："猪脚饭"、"驾照科目一"

  // === 上下文修饰（状语/定语） ===
  time_anchor: TimeAnchor | null;
  location: string | null;      // 地点："公司"、"公园"
  people: string[];             // 关系人：["小明", "同事"]
  mood: string | null;          // 心情修饰："开心"、"烦躁"
  energy: string | null;        // 能量状态："累"、"精力充沛"
  manner: string | null;        // 方式状语："匆忙地"、"认真地"

  // === 量化数据 ===
  cost: number | null;
  duration_minutes: number | null;
  metric: SemanticMetric | null;

  // === 关联意图 ===
  record_link_hint: string | null;  // 回溯搜索关键词，如"上周考试"
  item_hint: string | null;         // 推荐关联事项关键词
}

/** 复合句子单元间的关系类型 */
export type ClauseRelation = 'sequence' | 'contrast' | 'cause' | 'parallel';

/** 复合句拆分后的完整解析结果 */
export interface ParsedResult {
  is_compound: boolean;
  units: ParsedSemantic[];
  relations: Array<{
    from: number;
    to: number;
    type: ClauseRelation;
  }>;
  confidence: number; // 0~1，低于阈值时提示用户确认
}
```

```

```

```
ts  parsed_semantic: import('./semantic').ParsedSemantic | null;  // LLM 解析出的结构化语义

  people: string[] | null;           // 关系人数组
```

## Task 3: SQL 迁移脚本

​**文件**​: ﻿sql/009\_record\_semantic\_fields.sql﻿（新建）

sql

```
-- ============================================================
-- TETO 语义引擎 —— 记录表字段扩展
-- 文件：009_record_semantic_fields.sql
-- 说明：给 records 表新增语义解析相关字段
--       parsed_semantic 存储 LLM 解析的完整结构化 JSON
--       time_anchor_date / linked_record_id / location / people 为独立查询字段
-- ============================================================

-- 1. parsed_semantic —— LLM 解析出的完整语义 JSON
ALTER TABLE records ADD COLUMN IF NOT EXISTS parsed_semantic JSONB NULL;

-- 2. time_anchor_date —— 时间锚点解析后的目标日期
ALTER TABLE records ADD COLUMN IF NOT EXISTS time_anchor_date DATE NULL;

-- 3. linked_record_id —— 记录间关联（如"上周考试90分"关联到上周的考试记录）
ALTER TABLE records ADD COLUMN IF NOT EXISTS linked_record_id UUID NULL
  REFERENCES records(id) ON DELETE SET NULL;

-- 4. location —— 地点
ALTER TABLE records ADD COLUMN IF NOT EXISTS location TEXT NULL;

-- 5. people —— 关系人数组
ALTER TABLE records ADD COLUMN IF NOT EXISTS people TEXT[] NULL;
```
